const crypto = require('crypto');

class OrbitEcosystemClient {
  constructor(apiKey, apiSecret, hubUrl = 'https://orbitstaffing.io') {
    this.apiKey = apiKey;
    this.apiSecret = apiSecret;
    this.hubUrl = hubUrl;
  }

  sign(method, path, timestamp, body = '') {
    const payload = `${method}:${path}:${timestamp}:${body}`;
    return crypto.createHmac('sha256', this.apiSecret).update(payload).digest('hex');
  }

  async request(method, path, data = null) {
    const timestamp = Date.now().toString();
    const body = data ? JSON.stringify(data) : '';
    const signature = this.sign(method, path, timestamp, body);

    const response = await fetch(`${this.hubUrl}${path}`, {
      method,
      headers: {
        'Content-Type': 'application/json',
        'X-API-Key': this.apiKey,
        'X-API-Secret': signature,
        'X-Timestamp': timestamp,
      },
      body: data ? body : undefined,
    });
    return response.json();
  }

  async checkStatus() { return this.request('GET', '/api/ecosystem/status'); }
  async syncWorkers(workers) { return this.request('POST', '/api/ecosystem/sync/workers', { workers }); }
  async syncTimesheets(timesheets) { return this.request('POST', '/api/ecosystem/sync/timesheets', { timesheets }); }
  async getSnippets(category, language) { return this.request('GET', `/api/ecosystem/snippets?category=${category}&language=${language}`); }
  async pushSnippet(snippet) { return this.request('POST', '/api/ecosystem/snippets', snippet); }
  async log(event) { return this.request('POST', '/api/ecosystem/logs', event); }
}

module.exports = { OrbitEcosystemClient };
